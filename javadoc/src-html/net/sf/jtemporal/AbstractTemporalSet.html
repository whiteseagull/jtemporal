007    package net.sf.jtemporal;
008    
009    import java.util.Set;
010    import java.util.Iterator;
011    
012    /**
013     * Features common to the MultiMaps.
014     * @author Thomas A Beck
015     * @version $Id: AbstractTemporalSet.java,v 1.1 2004/05/02 21:40:30 tabeck Exp $
016     */
017    abstract public class AbstractTemporalSet implements TemporalSet {
018    
019      abstract public void clear();
020      abstract public int size(Object value);
021      abstract public boolean isEmpty();
022      abstract public Set valueSet(Instant instant);
023      abstract public Set valueSet();
024      abstract public Period getPeriod(Instant instant, Object value);
025      abstract public boolean contains(Instant instant, Object value);
026      abstract public boolean containsValue(Object value);
027      abstract public boolean put(Period p, Object value);
028      abstract public boolean remove(Period p, Object value);
029      abstract public void removeValue(Object value);
030      abstract public boolean removePeriod(Period p);
031      abstract public Set periodSet(Object value);
032      abstract public Instant firstInstant(Object value);
033      abstract public Instant lastInstant(Object value);
034      abstract public Period firstPeriod(Object value);
035      abstract public Period lastPeriod(Object value);
036      //abstract public TemporalSet subMap(Period p);
037    
038      public Period extent(Object value) {
039        return new Period(this.firstInstant(value), this.lastInstant(value));
040      }
041    
042      public boolean equals(Object o) {
043        throw new UnsupportedOperationException();
044      }
045    
046      public boolean putAll(TemporalSet mtm) {
047        if (mtm == null) throw new NullPointerException();
048        boolean ret = false;
049        for (Iterator v = mtm.valueSet().iterator(); v.hasNext(); ) {
050          Object value = v.next();
051          for (Iterator p = mtm.periodSet(value).iterator(); p.hasNext(); ) {
052            Period period = (Period) p.next();
053            ret |= this.put(period, value) ;
054          }
055        }
056        return ret;
057      }
058    
059    }
