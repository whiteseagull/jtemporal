007    package net.sf.jtemporal;
008    
009    import java.util.Set;
010    
011    /**
012     * Associates at any time an object (the map holder) to zero to many objects (values),
013     * for example: Employee-Skill. <BR>
014     * You can get the set of values valid at a given time by calling the method
015     * Set valueSet(Instant key).<BR>
016     * To add a mapping, you do not specify an Instant, but a Period specifying the validity
017     * of the object.    <BR><BR>
018     * <B>Note:</B> the Object value (passed to many methods) must have equals and hashCode
019     * defined consistently, as described in the java.lang.Object documentation.
020     * @author Thomas A Beck
021     * @version $Id: TemporalSet.java,v 1.1 2004/05/02 21:40:30 tabeck Exp $
022     */
023    public interface TemporalSet {
024    
025      /**
026       * Removes all the mappings from this map (optional operation).
027       * @throws UnsupportedOperationException if clear is not supported by this map.
028       */
029      void clear();
030    
031      /**
032       * Returns the number of Period-value mappings in this map.  If the
033       * map contains more than <tt>Integer.MAX_VALUE</tt> elements, returns
034       * <tt>Integer.MAX_VALUE</tt>.
035       * @return the number of Instant-value mappings in this map.
036       * Returns 0 when the value is unknown.
037       */
038      int size(Object value);
039    
040      /**
041       * Returns <tt>true</tt> if this map contains no Period-value mappings.
042       * @return <tt>true</tt> if this map contains no Period-value mappings.
043       */
044      boolean isEmpty();
045    
046      /**
047       * Returns a read-only Set containing the values defined in this map at the
048       * specified instant.
049       * Returns an empty set if the map contains no mapping at this Instant.
050       * @return a read-only Set containing the values defined in this map at the
051       * specified instant.
052       * @param instant instant whose associated value is to be returned
053       */
054      Set valueSet(Instant instant);
055    
056      /**
057       * Returns a read-only Set containing the values defined somewhen in this map.
058       * @return a read-only Set containing the values defined somewhen in this map.
059       * @param instant instant whose associated value is to be returned
060       */
061      Set valueSet();
062    
063      /**
064       * Returns the period of the mapping valid at the specified instant for the value.
065       * @return the period of the mapping valid at the specified instant for the value.
066       * Returns <tt>null</tt> if this map does not contain a mapping for the
067       * specified instant.
068       * @throws    NoSuchElementException if the value is unknown.
069       */
070      Period getPeriod(Instant instant, Object value);
071    
072      /**
073       * Returns <tt>true</tt> if there is a mapping for this value at the given instant.
074       * Same as containsValue(Instant instant, Object value)
075       * @param instant instant whose presence in this map is to be tested.
076       * @param value value whose presence in this map is to be tested.
077       * @return <tt>true</tt> if there is a mapping for this value at the given instant.
078       */
079      boolean contains(Instant instant, Object value);
080    
081    
082      /**
083       * Returns <tt>true</tt> if if ther is a mapping for the specified value.
084       * @param value value whose presence in this map is to be tested.
085       * @return <tt>true</tt> if this map maps one or more keys to the
086       *         specified value.
087       */
088      boolean containsValue(Object value);
089    
090      /**
091       * Associates the specified value with the specified Period in this map
092       * (optional operation).
093       * If the map previously contained a mapping to an equal value, during a period
094       * overlapping this period, the periods are merged.
095       *
096       * @param p period with which the specified value is to be associated.
097       * @param value value to be associated with the specified period.  Be careful 
098       * at the equals implemenation of the value object: the result of the equals 
099       * method is supposed to be immutable while it is stored in this collection.
100       * @return <tt>true</tt> if one or more previous values are overwritten
101       *         partialy or completely.  <tt>false</tt> if there is no conflicting
102       *         mapping.
103       *
104       * @throws UnsupportedOperationException if the <tt>put</tt> operation is
105       *              not supported by this map.
106       * @throws ClassCastException if the class of the specified value
107       *              prevents it from being stored in this map.
108       * @throws IllegalArgumentException if some aspect of this period or value
109       *              prevents it from being stored in this map.
110       * @throws NullPointerException if the period or the value is <tt>null</tt>
111       */
112      boolean put(Period p, Object value);
113      
114    
115      /**
116       * Removes the mapping(s) for this period and value from this map if present
117       * (optional operation).
118       *
119       * @param p period whose mappings are to be removed from the map.
120       * @param value value whose mapping are to be removed from the map.
121       * @return <tt>true</tt> if one or more previous values are removed
122       *         partialy or completely.  <tt>false</tt> if there is no mapping changed.
123       * @throws UnsupportedOperationException if the <tt>remove</tt> method is
124       *         not supported by this map.
125       * @throws    NoSuchElementException if the value is unknown.
126       */
127      boolean remove(Period p, Object value);
128    
129      /**
130       * Removes all the mapping(s) for this value from this map if present (optional
131       * operation).
132       *
133       * @param value value whose mapping are to be removed from the map.
134       * @throws UnsupportedOperationException if the <tt>remove</tt> method is
135       *         not supported by this map.
136       * @throws    NoSuchElementException if the value is unknown.
137       */
138      void removeValue(Object value);
139    
140      /**
141       * Removes all the mapping(s) for this period from this map if present (optional
142       * operation).
143       *
144       * @param p period whose mappings are to be removed from the map.
145       * @return <tt>true</tt> if one or more previous values are removed
146       *         partialy or completely.  <tt>false</tt> if there is any mapping changed.
147       * @throws UnsupportedOperationException if the <tt>remove</tt> method is
148       *         not supported by this map.
149       */
150      boolean removePeriod(Period p);
151    
152      /**
153       * Copies all of the mappings from the specified map to this map
154       * (optional operation).  These mappings will replace any mappings that
155       * this map had for any of the periods currently in the specified map.
156       *
157       * @param tm Mappings to be stored in this map.
158       *
159       * @return <tt>true</tt> if one or more previous values are overwritten
160       *         partialy or completely.  <tt>false</tt> if there is any conflicting
161       *         mapping.
162       *
163       * @throws UnsupportedOperationException if the <tt>putAll</tt> method is
164       *              not supported by this map.
165       *
166       * @throws ClassCastException if the class of a value in the
167       *              specified map prevents it from being stored in this map.
168       *
169       * @throws IllegalArgumentException some aspect of a value in the
170       *              specified map prevents it from being stored in this map.
171       */
172      boolean putAll(TemporalSet mtm);
173    
174    
175      /**
176       * Returns a set view of the periods contained in this map for the give value.
177       * The set is
178       * backed by the map, so changes to the map are reflected in the set, and
179       * vice-versa.  If the map is modified while an iteration over the set is
180       * in progress, the results of the iteration are undefined.  The set
181       * supports element removal, which removes the corresponding mapping from
182       * the map, via the <tt>Iterator.remove</tt>, <tt>Set.remove</tt>,
183       * <tt>removeAll</tt> <tt>retainAll</tt>, and <tt>clear</tt> operations.
184       * It does not support the add or <tt>addAll</tt> operations.
185       *
186       * @return a set view of the periods contained in this map for the give value.
187       * @throws    NoSuchElementException if the value is unknown.
188       */
189      Set periodSet(Object value);
190    
191    
192      /**
193       * Returns the first (lowest) instant currently defined for the given value.
194       *
195       * @return the first (lowest) instant currently defined for the given value.
196       * @throws    NoSuchElementException if the value is unknown.
197       */
198      Instant firstInstant(Object value);
199    
200      /**
201       * Returns the last (highest) instant currently defined for the given value.
202       *
203       * @return the last (highest) instant currently defined for the given value.
204       * @throws    NoSuchElementException if the value is unknown.
205       */
206      Instant lastInstant(Object value);
207    
208      /**
209       * Returns the first (lowest) period currently defined for the given value.
210       *
211       * @return the first (lowest) period currently defined for the given value.
212       * @throws    NoSuchElementException if the value is unknown.
213       */
214      Period  firstPeriod(Object value);
215    
216      /**
217       * Returns the last (highest) period currently defined for the given value.
218       *
219       * @return the last (highest) period currently defined for the given value.
220       * @throws    NoSuchElementException if the value is unknown.
221       */
222      Period  lastPeriod(Object value);
223    
224      /**
225       * Return a period enclosing firstInstant(value) and lastInstantvalue()
226       * @return new Period(firstInstant(value), lastInstant(value))
227       * @throws    NoSuchElementException if the value is unknown.
228       * @see TemporalSet#firstInstant(Object)
229       * @see TemporalSet#lastInstant(Object)
230       */
231      Period  extent(Object value);
232    
233      /**
234       * Returns a TemporalAttribute which is either a copy or a view of this map,
235       * restricted to the given period.    (optional operation).
236       * @throws UnsupportedOperationException if clear is not supported by this map.
237         TemporalSet subMap(Period p); */
238    }
