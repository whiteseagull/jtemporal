007    package net.sf.jtemporal.util;
008    
009    import java.util.*;
010    import java.io.Serializable;
011    
012    
013    /**
014     * Set proxy, restricting operations to read-only.  <BR>
015     * <B>For methods description, see java.util.Set</B>
016     * @author Thomas Beck
017     * @version $Id: ReadOnlySet.java,v 1.2 2004/11/01 11:37:48 tabeck Exp $
018     */
019    
020    public class ReadOnlySet implements Set, Serializable {
021    
022      private static final String ERROR_MSG = "This Set is read-only";
023      private static final long serialVersionUID = 9826560106639L;
024    
025      private final Set remoteSet;
026    
027      public ReadOnlySet(Set s) {
028        this.remoteSet = s;
029      }
030      public boolean contains(Object o) {
031        return remoteSet.contains(o);
032      }
033      public boolean remove(Object o) {
034        throw new UnsupportedOperationException(ERROR_MSG);
035      }
036      public Object[] toArray(Object[] parm1) {
037        return remoteSet.toArray(parm1);
038      }
039      public boolean isEmpty() {
040        return remoteSet.isEmpty();
041      }
042      public int size() {
043        return remoteSet.size();
044      }
045      public boolean containsAll(Collection c) {
046        return remoteSet.containsAll(c);
047      }
048      public Object[] toArray() {
049        return remoteSet.toArray();
050      }
051    
052    
053      public boolean retainAll(Collection parm1) {
054        throw new UnsupportedOperationException(ERROR_MSG);
055      }
056      public boolean add(Object o) {
057        throw new UnsupportedOperationException(ERROR_MSG);
058      }
059      public void clear() {
060        throw new UnsupportedOperationException(ERROR_MSG);
061      }
062      public Iterator iterator() {
063        return new Iterator() {
064          private final Iterator remoteIterator = ReadOnlySet.this.remoteSet.iterator();
065          public boolean hasNext() {
066            return remoteIterator.hasNext();
067          }
068          public Object next() {
069            return remoteIterator.next();
070          }
071          public void remove() {
072            throw new UnsupportedOperationException(ERROR_MSG);
073          }
074        };
075      }
076      public boolean addAll(Collection parm1) {
077        throw new UnsupportedOperationException(ERROR_MSG);
078      }
079      public boolean removeAll(Collection parm1) {
080        throw new UnsupportedOperationException(ERROR_MSG);
081      }
082    }
