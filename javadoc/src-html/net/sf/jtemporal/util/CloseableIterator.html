008    package net.sf.jtemporal.util;
009    
010    import java.util.Iterator;
011    
012    /**
013     * An iterator the should be closed as soon it is not used anymore. <br>
014     * This is useful to avoid waiting the garbage collection to free resources, 
015     * such as database resources.
016     * @author Thomas A Beck
017     * @version $Id: CloseableIterator.java,v 1.1 2004/11/01 11:37:48 tabeck Exp $
018     */
019    public interface CloseableIterator extends Iterator {
020            
021            /**
022             * Closes the iterator. <br>
023             * Once this method has been called, the iterator becomes unusable. <br>
024             * If the cursor is already closed, this method just returns.
025             */
026            void    close();
027            
028            /**
029             * Checks whether the iterator is open, or already closed.
030             * @return <code>true</code> if the cursor is open.
031             */     
032            boolean isOpen();
033    }
