007    package net.sf.jtemporal.util;
008    
009    import java.util.Iterator;
010    import java.util.NoSuchElementException;
011    import java.io.InvalidObjectException;
012    import java.io.Serializable;
013    
014    
015    /**
016     * Immutable empty read-only iterator (singleton).  <BR>
017     * <B>For methods description, see java.util.Iterator</B>
018     * @author Thomas Beck
019     * @version $Id: EmptyIterator.java,v 1.1 2003/11/16 18:07:01 tabeck Exp $
020     */
021    
022    public class EmptyIterator implements Serializable, Iterator {
023    
024      private static final EmptyIterator INSTANCE = new EmptyIterator();
025      private static final long serialVersionUID = 9365825104636L;
026    
027      static public EmptyIterator getInstance() {
028        return INSTANCE;
029      }
030    
031      private EmptyIterator() {
032      }
033    
034      /**
035       * Resolves instances being deserialized to the predefined unique instance.
036       */
037      private Object readResolve() throws InvalidObjectException {
038        return INSTANCE;
039      }
040    
041      public boolean hasNext() {
042        return false;
043      }
044      public Object next() {
045        throw new NoSuchElementException();
046      }
047      public void remove() {
048        throw new java.lang.UnsupportedOperationException();
049      }
050    }
