007    package net.sf.jtemporal.util;
008    
009    import java.util.*;
010    import java.io.InvalidObjectException;
011    import java.io.Serializable;
012    
013    /**
014     * Immutable empty read-only set (singleton).  <BR>
015     * <B>For methods description, see java.util.Set</B>
016     * @author Thomas Beck
017     * @version $Id: EmptySet.java,v 1.1 2003/11/16 18:07:01 tabeck Exp $
018     */
019    public class EmptySet extends AbstractSet implements Serializable {
020    
021      private static final EmptySet INSTANCE = new EmptySet();
022      private static final long serialVersionUID = 1896628674035L;
023    
024      static public EmptySet getInstance() {
025        return INSTANCE;
026      }
027    
028      static final private String IMMUTABLE_ERROR = "This EmptySet is immutable";
029      static private Object[] emptyArray = new Object[0];
030    
031      private EmptySet() {
032      }
033    
034      /**
035       * Resolves instances being deserialized to the predefined constant.
036       */
037      private Object readResolve() throws InvalidObjectException {
038        return INSTANCE;
039      }
040    
041      public boolean contains(Object o) {
042        return false;
043      }
044      public boolean remove(Object o) {
045        throw new UnsupportedOperationException(IMMUTABLE_ERROR);
046      }
047      public boolean isEmpty() {
048        return true;
049      }
050      public boolean add(Object o) {
051        throw new UnsupportedOperationException(IMMUTABLE_ERROR);
052      }
053      public int size() {
054        return 0;
055      }
056      public boolean containsAll(Collection parm1) {
057        return false;
058      }
059      public Object[] toArray() {
060        return EmptySet.emptyArray;
061      }
062      public void clear() {
063        throw new UnsupportedOperationException(IMMUTABLE_ERROR);
064      }
065      public Iterator iterator() {
066        return EmptyIterator.getInstance();
067      }
068      public boolean addAll(Collection parm1) {
069        throw new UnsupportedOperationException(IMMUTABLE_ERROR);
070      }
071    }
