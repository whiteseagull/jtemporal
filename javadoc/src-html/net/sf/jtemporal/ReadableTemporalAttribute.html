008    package net.sf.jtemporal;
009    
010    /**
011     * Superinterface common to anything able to return an object for a given instant.
012     * @author Thomas A Beck
013     * @version $Id: ReadableTemporalAttribute.java,v 1.1 2004/05/02 15:51:00 tabeck Exp $
014     */
015    public interface ReadableTemporalAttribute {
016            
017            /**
018             * Returns the object valid at the given instant.
019             * @param i instant to query
020             * @return the object valid at the given instant
021             */
022            public Object get(Instant i);
023    }
