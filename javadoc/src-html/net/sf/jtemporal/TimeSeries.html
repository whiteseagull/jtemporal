007    package net.sf.jtemporal;
008    
009    /**
010     * This is more or less a Map, whose key is an Instant. <BR>
011     * <b>Warning, some more methods will be added later.</b> 
012     * especially given that for the moment no sorting-related methods 
013     * are present (iterators, etc).  
014     * @author Thomas A Beck
015     * @version $Id: TimeSeries.java,v 1.5 2004/11/01 11:37:47 tabeck Exp $
016     * @todo add more methods
017     */
018    public interface TimeSeries extends ReadableTemporalAttribute {
019            
020            /**
021             * Creates a new entry in this timeseries.
022             * @param i the instant whom the value must be associated to.  Must be not null.
023             * @param value the value.  Must be not null.
024             * @throws UnsupportedOperationException if the TimeSeries is read-only
025             * @return the value previously associated to the i instant, or null.
026             */
027            public Object put(Instant i, Object value);
028            
029            /**
030             * Gets the value currently associated to the i instant.
031             * @param the instant, key for this value.
032             * @return the value currently associated to the i instant, or null.
033             */
034            public Object get(Instant i); // from ReadableTemporalAttribute
035            
036            /**
037             * Removes an entry from this timeseries.
038             * @param i the instant to be removed. Must be not null.
039             * @return the value that was previously associated to the i instant
040             * @throws UnsupportedOperationException if the TimeSeries is read-only
041             */
042            public Object remove(Instant i);
043            
044            /**
045             * Remove a whole period from the timeseries (optional implementation). 
046             * @param p the period enclosing the entries to be removed.
047             * @throws UnsupportedOperationException if the TimeSeries is read-only
048             * @return <code>true</code> if something has been actually removed
049             */     
050            public void remove(Period p);
051            
052            /**
053             * Returns <tt>true</tt> if this TimeSeries contains a mapping for the specified
054             * Instant.
055             * @param i instant whose presence in this TimeSeries is to be tested.
056             * @return <tt>true</tt> if this TimeSeries contains a mapping for the specified
057             * instant.
058             */
059            public boolean containsInstant(Instant i);
060            
061            /**
062            * Returns a Set view of the Instants (keys) contained in this TimeSeries.  
063            * The set's iterator will return the keys in ascending order.  
064            * The set is backed by this <tt>TimeSeries</tt> instance, 
065            * so changes to this TimeSeries are reflected in the Set, and vice-versa.  
066            * The Set supports element removal, which removes the corresponding mapping 
067            * from the map, via the <tt>Iterator.remove</tt>, <tt>Set.remove</tt>, <tt>removeAll</tt>,
068            * <tt>retainAll</tt>, and <tt>clear</tt> operations.  It does not support
069            * the <tt>add</tt> or <tt>addAll</tt> operations.
070            *
071            * @return a set view of the instants that are keys in this TimeSeries.
072            */
073            //public Set instantSet();
074    }
