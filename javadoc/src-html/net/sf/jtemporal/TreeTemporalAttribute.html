009    
010    import net.sf.jtemporal.spi.TreeTemporalAttributeStorage;
011    
012    /**
013     * Uses internally a TreeMap to store the mappings.
014     * The Periods as used as keys.
015     * <BR>
016     * Links at any time an object (the map holder) to zero or one object (value),
017     * example: Employee-Salary.
018     * You get the value valid at a given time with the method Object get(Instant i).
019     * To add a mapping, you do not specify an Instant, but a Period specifying the validity
020     * of the object.  If another object exists valid during this period, the validity will be
021     * overwritten for new object period.
022     * <BR><b>
023     * Note that this implementation is not synchronized.</b> If multiple
024     * threads access a map concurrently, and at least one of the threads modifies
025     * the map structurally, it <i>must</i> be synchronized externally.  (A
026     * structural modification is any operation that adds or deletes one or more
027     * mappings; merely changing the value associated with an existing key is not
028     * a structural modification.)  This is typically accomplished by
029     * synchronizing on some object that naturally encapsulates the map.  If no
030     * such object exists, the map should be "wrapped" using the
031     * <tt>Collections.synchronizedMap</tt> method.  This is best done at creation
032     * time, to prevent accidental unsynchronized access to the map:
033     * <pre>
034     *     TemporalAttribute tm = SynchronizedTemporalMap.getInstance(new TreeTemporalSet(...));
035     * </pre><p>
036     * <br>
037     * Actually, this is just a TemporalAttributeImpl instance using a 
038     * TreeTemporalAttributeStorage as a storage.
039     * @author Thomas A Beck
040     * @version $Id$
041     * @see net.sf.jtemporal.TemporalAttributeImpl
042     * @see net.sf.jtemporal.spi.TreeTemporalAttributeStorage
043     */
044    public class TreeTemporalAttribute extends TemporalAttributeImpl {
045    
046            public TreeTemporalAttribute() {
047                    super(new TreeTemporalAttributeStorage());
048            }
049    
050    }
