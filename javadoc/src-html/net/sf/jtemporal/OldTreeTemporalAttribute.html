011    /**
012     * Uses internally a TreeMap to store the mappings.
013     * The Periods as used as keys.
014     * <BR>
015     * Links at any time an object (the map holder) to zero or one object (value),
016     * example: Employee-Salary.
017     * You get the value valid at a given time with the method Object get(Instant i).
018     * To add a mapping, you do not specify an Instant, but a Period specifying the validity
019     * of the object.  If another object exists valid during this period, the validity will be
020     * overwritten for new object period.
021     * <BR><b>
022     * Note that this implementation is not synchronized.</b> If multiple
023     * threads access a map concurrently, and at least one of the threads modifies
024     * the map structurally, it <i>must</i> be synchronized externally.  (A
025     * structural modification is any operation that adds or deletes one or more
026     * mappings; merely changing the value associated with an existing key is not
027     * a structural modification.)  This is typically accomplished by
028     * synchronizing on some object that naturally encapsulates the map.  If no
029     * such object exists, the map should be "wrapped" using the
030     * <tt>Collections.synchronizedMap</tt> method.  This is best done at creation
031     * time, to prevent accidental unsynchronized access to the map:
032     * <pre>
033     *     TemporalAttribute tm = SynchronizedTemporalMap.getInstance(new TreeTemporalSet(...));
034     * </pre><p>
035     * @author Thomas A Beck
036     * @version $Id: TreeTemporalAttribute.java,v 1.5 2004/11/01 11:37:47 tabeck Exp $
037     *
038     */
039

131      * // GET_PERIOD
132      */
133      public Period getPeriod(Instant instant) {
134    
135        SortedMap headMap = this.map.headMap(instant);  //headMap < key
136        if (!headMap.isEmpty()) {
137          Period prev = (Period) headMap.lastKey();
138          if (prev.contains(instant)) return prev;
139        }
140    
141        SortedMap tailMap = this.map.tailMap(instant);  //tailMap >= key
142        if (!tailMap.isEmpty()) {
143          Period next = (Period) tailMap.firstKey();
144          if (next.contains(instant)) return next;
145        }
146    
147        return null;  // no period found containing this instant
148      }

166      public boolean put(Period p, Object value) {
167        if (value == null) throw new NullPointerException();
168    
169        if (this.valueType != null) {
170          if (! this.valueType.isInstance(value) ) {
171            throw new ClassCastException();
172          }
173        }
174    
175        boolean ret = this.remove(p);
176        //this.versionNumber++; done by remove(p)
177        Object o = map.put(p, value);
178        if (o != null)  throw new IllegalStateException();  // should never happen
179    
180        return ret;
181      }

185      public boolean remove(Period p) {
186        if (p == null) throw new NullPointerException();
187    
188        this.versionNumber++;
189        boolean removed = false;
190        Instant start = p.getStart();
191        Instant end = p.getEnd();
192    
193        //first overlapping period
194        Period firstPeriod = this.getPeriod(start);
195        if (firstPeriod != null) {
196          removed = true;
197    
198              Object value = map.remove(firstPeriod);
199    
200          if (firstPeriod.getStart().compareTo(start) < 0) {  // if startPeriod.start < start
201            map.put(new Period(firstPeriod.getStart(), start), value);
202          }
203          
204          if (firstPeriod.getEnd().compareTo(end) > 0) { // if startPeriod.end > end
205                    map.put(new Period(end, firstPeriod.getEnd()), value);          
206          }
207        }
208    
209        //other overlapping periods
210        Iterator i = this.map.subMap(start, end).keySet().iterator();
211        while (i.hasNext()) {
212          removed = true;
213          Period act = (Period) i.next();
214          if (act.getEnd().compareTo(end) <= 0) {
215            i.remove();
216          } else {
217                    if (i.hasNext()) {
218                            // this can never happen, unless there is a bug in my code
219                            throw new IllegalStateException();
220                    }
221            Object value = map.remove(act);
222            map.put(new Period(end, act.getEnd()), value);
223            break;
224          }
225        }
226    
227        return removed;
228            
229      }

480      /**
481       * Compares Periods and Instants by using the Period start instant.
482       * Storing instants in the TreeMap would have been more logical, but that would
483       * have needed an additional object containing period+value.
484       * The performace difference is not clear, but it's clear the code is more simple in this way.
485       */
486      static private class TreeComparator implements Comparator
487      {
488        /**
489         * If both objects are Periods, they are compared as periods
490         * otherwise they are compared as instants
491         * (taking the start instant if one is a Period).
492         * It would be more OO to implement a common method in Period and
493         * in Instant, but I want to keep the Instant interface as simple as possible.
494         */
495        public int compare(Object o1, Object o2) {
496          Instant i1, i2;
497    
498          if (o1 instanceof Period) {
499                    if (o2 instanceof Period)
500                      return ((Period) o1).compareTo(o2);
501                    else
502                      i1 = ((Period) o1).getStart();
503                  } else { // o1 instanceof Instant
504                    i1 = (Instant) o1;
505                    if (o2 instanceof Period)
506                      o2 = ((Period) o2).getStart();
507          } // if
508    
509          i2 = (Instant) o2;
510          return i1.compareTo(i2);
511    
512    /*      if (o1 instanceof Period && o2 instanceof Period)
513            return ((Period) o1).compareTo(o2);
514    
515          Instant i1 = o1 instanceof Instant ? (Instant) o1 : ((Period) o1).getStart();
516          Instant i2 = o2 instanceof Instant ? (Instant) o2 : ((Period) o2).getStart();
517          return i1.compareTo(i2);
518    */
519        }
520      }
