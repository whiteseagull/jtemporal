007    package net.sf.jtemporal;
008    /**
009     * You must implement this interface (basically a Comparable) to
010     * represent time. The implementing object must be immutable.  <BR>
011     * You can write a little adapter wrapping java.util.Date.  Most users rether use
012     * "pure" dates, to avoid issues with timezones. There are many opensource
013     * implementations of Julian date and of Gregorian date.
014     * Many companies have their own date/calendar framework.
015     * For example, in my unit test classes, I have simply adapted an int. <BR>
016     * <B>Any object implementing this interface is required to be immutable.  This
017     * is part of the contract.</B><BR>
018     * Do not forget to implement equals and hashCode properly, as defined in the
019     * java.lang.Object documentation.
020     * Also, the equals method must be consistent with the compareTo method.
021     * @stereotype immutable
022     * @author Thomas A Beck
023     * @version $Id: Instant.java,v 1.3 2004/11/01 11:37:47 tabeck Exp $
024     */
025    public interface Instant extends Comparable
026    {
027      /**
028       * Compares this object with the specified object for order.  Returns a
029       * negative integer, zero, or a positive integer as this object is less
030       * than, equal to, or greater than the specified object.<p>
031       *
032       * The implementor must ensure <tt>sgn(x.compareTo(y)) ==
033       * -sgn(y.compareTo(x))</tt> for all <tt>x</tt> and <tt>y</tt>.  (This
034       * implies that <tt>x.compareTo(y)</tt> must throw an exception iff
035       * <tt>y.compareTo(x)</tt> throws an exception.)<p>
036       *
037       * The implementor must also ensure that the relation is transitive:
038       * <tt>(x.compareTo(y)>0 && y.compareTo(z)>0)</tt> implies
039       * <tt>x.compareTo(z)>0</tt>.<p>
040       *
041       * Finally, the implementer must ensure that <tt>x.compareTo(y)==0</tt>
042       * implies that <tt>sgn(x.compareTo(z)) == sgn(y.compareTo(z))</tt>, for
043       * all <tt>z</tt>.<p>
044       *
045       * <B>It is required that <tt>(x.compareTo(y)==0) == (x.equals(y))</tt>.</B>
046       *
047       * @param   o the Object to be compared.
048       * @return  a negative integer, zero, or a positive integer as this object
049       *            is less than, equal to, or greater than the specified object.
050       *
051       * @throws ClassCastException if the specified object's type prevents it
052       *         from being compared to this Object.
053       */
054      public int compareTo(Object o);
055    
056      /**
057       * Returns the biggest existing Instant.                  <BR><tt>
058       * positiveInfinity().compareTo(positiveInfinity()) == 0  <BR>
059       * positiveInfinity().equals(positiveInfinity()) == true  <BR></tt>
060       * The implementation of this method is not mandatory.    <BR>
061       * It could become mandatory in a future version.
062       * @throws UnsupportedOperationException if this method is not implemented
063       * @return the instant which is greater than any other instant
064       */
065      Instant positiveInfinity();
066    
067      /**
068       * Returns the lowest existing Instant.                   <BR><tt>
069       * positiveInfinity().compareTo(positiveInfinity()) == 0  <BR>
070       * positiveInfinity().equals(positiveInfinity()) == true  <BR></tt>
071       * The implementation of this method is not mandatory.    <BR>
072       * It could become mandatory in a future version.
073       * @throws UnsupportedOperationException if this method is not implemented
074       * @return the instant which is lower than any other instant
075       */
076      Instant negativeInfinity();
077    
078    }
